```python
from fastapi import FastAPI
from core import (
    # Interceptors
    get_request_interceptor, InterceptorType,
    # Resource Pool
    create_resource_pool, PoolStatus,
    # Advanced Caching
    get_advanced_cache_strategy, CacheStrategy,
    # Todo lo anterior...
)

app = FastAPI()

# Request Interceptor
interceptor = get_request_interceptor()
interceptor.register_interceptor(
    InterceptorType.PRE_REQUEST,
    lambda req: {**req, "timestamp": datetime.now()},
    priority=10
)

# Resource Pool
def create_db_connection():
    return DatabaseConnection()

pool = create_resource_pool(
    factory=create_db_connection,
    min_size=5,
    max_size=20
)
resource = await pool.acquire()
# usar resource
await pool.release(resource)

# Advanced Caching
cache_strategy = get_advanced_cache_strategy(
    cache_service=redis_cache,
    strategy=CacheStrategy.WRITE_THROUGH
)
value = await cache_strategy.get("key", loader=lambda: fetch_from_db())
await cache_strategy.set("key", value, writer=lambda k, v: save_to_db(k, v))
```
