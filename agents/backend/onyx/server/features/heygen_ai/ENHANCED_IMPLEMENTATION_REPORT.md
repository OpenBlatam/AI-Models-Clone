```python
# Configurar ElevenLabs
config = ServiceConfig(
    service_type=ServiceType.VOICE_SYNTHESIS,
    name="elevenlabs",
    api_key="your-api-key",
    base_url="https://api.elevenlabs.io/v1"
)

service = ElevenLabsService(config)
manager = ExternalAPIManager()
manager.register_service(service)

# Generar voz premium
audio_data = await service.generate_speech(
    text="Hello, this is premium voice synthesis!",
    voice_id="premium_voice_id"
)
```

```python
    # Procesamiento optimizado
    return await generate_avatar(prompt)
```
