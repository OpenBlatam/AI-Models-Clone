### Error Handling Testing
```python
from tests.test_error_handling_helpers import (
    create_mock_error_handler,
    assert_error_handled,
    classify_error
)

handler = create_mock_error_handler()
handler.handle(ValueError("Test error"))
assert_error_handled(handler, ValueError)

classification = classify_error(ValueError("Test"))
assert classification["severity"] == "error"
```

### Pagination Testing
```python
from tests.test_pagination_helpers import (
    create_paginated_response,
    assert_pagination_valid
)

items = list(range(100))
response = create_paginated_response(items, page=1, page_size=10)
assert_pagination_valid(response, expected_page=1, expected_page_size=10)
```

```python
from tests.test_search_helpers import (

engine = create_mock_search_engine({"test": ["result1", "result2"]})

```
