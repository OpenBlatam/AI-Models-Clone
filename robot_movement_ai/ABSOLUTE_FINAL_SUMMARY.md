## 🚀 Quick Start Final

### Desarrollo

```bash
# Setup completo
make setup

# Tests
make test

# Run
make run

# Benchmark
python scripts/benchmark.py

# Health Monitor
python scripts/health_monitor.py
```

### Producción

```bash
# Docker
docker-compose up -d

# Kubernetes
kubectl apply -f k8s/deployment.yaml

# Load Test
python scripts/load_test.py --users 100 --duration 5m
```

### Uso de Características

```python
# GraphQL
# Query en http://localhost:8010/graphql
query {
  robots {
    id
    name
    status
  }
}

# Batch Processing
from core.architecture.batch_processor import create_batch_processor
processor = create_batch_processor(chunk_size=100, max_workers=10)
result = await processor.process_batch(items, process_item)

# Job Queue
from core.architecture.job_queue import enqueue_job, JobPriority
job_id = enqueue_job(my_function, arg1, priority=JobPriority.HIGH)

# Throttling
from core.architecture.throttling import create_throttler, ThrottleAlgorithm
throttler = create_throttler(algorithm=ThrottleAlgorithm.SLIDING_WINDOW)
allowed, remaining, reset = throttler.is_allowed("user-1", 100, 60)

# Audit Logging
from core.architecture.audit_log import audit_log, AuditAction
@audit_log(AuditAction.EXECUTE, "robot")
async def move_robot(robot_id: str):
    ...
```
